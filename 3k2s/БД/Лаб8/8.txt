CREATE DEFINER=`root`@`localhost` PROCEDURE `задание_11`(IN Price INT, IN Count INT)
BEGIN
SELECT b.НаимТовара `Наименование`, a.Количество `Кол-во`, b.Цена1*Price `Цена`
FROM Запасы a, Товары b
WHERE (b.КодТовара = a.КодТовара
	AND a.Количество < Count)
ORDER BY b.НаимТовара;
END
